// Set RGB directly
// RGB values are in the range 0%..100%

{"color":{"r":20,"g":30,"b":60}}

// You can also send white1/2 in the JSON command

{"color":{"r":20,"g":30,"b":60,"w1":50,"w2":80}}


// Set HSV values
// HSV values are in the range 0..255

{"hsv":{"h":0,"s":255,"v":128}}   // red with 50% brightness
{"hsv":{"h":0,"s":255,"v":255}}   // Full red
{"hsv":{"h":128,"s":255,"v":255}} // Full green
